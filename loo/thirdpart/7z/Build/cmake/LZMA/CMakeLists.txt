SET(LIB_NAME LZMA)

SET(C_FILES
	${SEVENZIP_PROJECT_DIR}/C/Alloc.c
	${SEVENZIP_PROJECT_DIR}/C/LzFind.c
	${SEVENZIP_PROJECT_DIR}/C/LzFindMt.c
	${SEVENZIP_PROJECT_DIR}/C/LzmaDec.c
	${SEVENZIP_PROJECT_DIR}/C/LzmaEnc.c
	${SEVENZIP_PROJECT_DIR}/C/LzmaLib.c
)
IF(WIN32)
	SET(C_FILES ${C_FILES}
		${SEVENZIP_PROJECT_DIR}/C/Util/LzmaLib/LzmaLibExports.c
	)
ENDIF()

SET(DEF_FILES
	${SEVENZIP_PROJECT_DIR}/C/Util/LzmaLib/LzmaLib.def
)
SET(LZMA_HEADER_FILES
	${SEVENZIP_PROJECT_DIR}/C/Alloc.h
	${SEVENZIP_PROJECT_DIR}/C/LzFind.h
	${SEVENZIP_PROJECT_DIR}/C/LzFindMt.h
	${SEVENZIP_PROJECT_DIR}/C/LzHash.h
	${SEVENZIP_PROJECT_DIR}/C/LzmaDec.h
	${SEVENZIP_PROJECT_DIR}/C/LzmaEnc.h
	${SEVENZIP_PROJECT_DIR}/C/LzmaLib.h
	${SEVENZIP_PROJECT_DIR}/C/Threads.h
	${SEVENZIP_PROJECT_DIR}/C/Types.h
)
SOURCE_GROUP("Source Files" FILES ${DEF_FILES} ${C_FILES})
SOURCE_GROUP("Header Files" FILES ${LZMA_HEADER_FILES})

ADD_DEFINITIONS(-DNO_ASM)
ADD_DEFINITIONS(-DLZMALIB_EXPORTS)
ADD_DEFINITIONS(-D_7ZIP_ST)

ADD_LIBRARY(${LIB_NAME} ${LOO_PREFERRED_LIB_TYPE}
	${C_FILES} ${DEF_FILES} ${LZMA_HEADER_FILES}
)

SET(LIB_OUTPUT_NAME ${LIB_NAME})

SET_TARGET_PROPERTIES(${LIB_NAME} PROPERTIES
	ARCHIVE_OUTPUT_DIRECTORY ${SEVENZIP_OUTPUT_DIR}
	ARCHIVE_OUTPUT_DIRECTORY_DEBUG ${SEVENZIP_OUTPUT_DIR}
	ARCHIVE_OUTPUT_DIRECTORY_RELEASE ${SEVENZIP_OUTPUT_DIR}
	ARCHIVE_OUTPUT_DIRECTORY_RELWITHDEBINFO ${SEVENZIP_OUTPUT_DIR}
	ARCHIVE_OUTPUT_DIRECTORY_MINSIZEREL ${SEVENZIP_OUTPUT_DIR}
	PROJECT_LABEL ${LIB_NAME}
	DEBUG_POSTFIX ${CMAKE_DEBUG_POSTFIX}
	OUTPUT_NAME ${LIB_OUTPUT_NAME}${LOO_OUTPUT_SUFFIX}
)
IF(LOO_PLATFORM_DARWIN OR LOO_PLATFORM_LINUX)
	SET_TARGET_PROPERTIES(${LIB_NAME} PROPERTIES
		LIBRARY_OUTPUT_DIRECTORY ${LOO_BIN_DIR}
		LIBRARY_OUTPUT_DIRECTORY_DEBUG ${LOO_BIN_DIR}
		LIBRARY_OUTPUT_DIRECTORY_RELEASE ${LOO_BIN_DIR}
		LIBRARY_OUTPUT_DIRECTORY_RELWITHDEBINFO ${LOO_BIN_DIR}
		LIBRARY_OUTPUT_DIRECTORY_MINSIZEREL ${LOO_BIN_DIR})
ENDIF()

TARGET_LINK_LIBRARIES(${LIB_NAME})

IF(LOO_PLATFORM_WINDOWS)
	INSTALL(TARGETS ${LIB_NAME}
		RUNTIME DESTINATION ${LOO_BIN_DIR}
		LIBRARY DESTINATION ${LOO_BIN_DIR}
	)
ENDIF()

SET_TARGET_PROPERTIES(${LIB_NAME} PROPERTIES FOLDER "7z")
